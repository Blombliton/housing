import pandas as pd
data = pd.read_csv('housing.csv', sep = ',')
data = data.sort_values(by='longitude', ascending=False)[:1000]
data
__________________________________________________________________________________________

import matplotlib.pyplot as plt
fig, ax = plt.subplots()
data.plot(kind='bar', x= "total_rooms", y= "total_bedrooms", ax = ax)

avg = data['total_bedrooms'].mean()
competitors = data[data.total_bedrooms > avg]
ax.axhline(y=len(competitors)-0.5, color='g', label='Average', linestyle='-', linewidth=4)
ax.legend().set_visible(False) 

fig.savefig('total_bedrooms.png', transparent=False, dpi=80, bbox_inches="tight")
